プログラム
int[]x=new int[100];
int[]y=new int[100];
int i=0;

void setup(){
size(500,500);
}

void draw(){
int j,k;
background(255);
if(i>0){
for(j=0;j<=i-2;j=j+2){
line(x[j],y[j],x[j+1],y[j+1]);
}
}
if(i>=1){
if(i%2==1){
line(x[i-1],y[i-1],mouseX,mouseY);
}
}
if(keyPressed==true){
if(key=='s'){
println("start");
x[i]=mouseX;
y[i]=mouseY;
i=i+1;
delay(800);
}
else if(key=='f'){
println("finish");
x[i]=mouseX;
y[i]=mouseY;
i=i+1;
delay(800);
}
else if(key=='g'){
for(k=0;k<=i-1;k=k+2){
println("line(",x[k],",",y[k],",",x[k+1],",",y[k+1],");");
}
delay(1000);
}
}
}
使い方
カーソルをはじめの位置に持っていき、キーボードのSを押す。
※この時、コンソールにstartと書かれているかチェック！！
線が出てきたら終わりの位置にカーソルを持っていき、キーボードのFを押す。
※この時、コンソールにfinishと書かれているかチェック！！
線が全て描けたらキーボードのGを押す
そうするとコンソール上にたくさんのline()関数が表示されるので、コピペ
以上！！！！！^_^
